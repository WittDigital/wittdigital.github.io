async function updateBotStatus() { const SERVER_ID = '1330733636219043961'; // 你剛才測試成功的 Server ID const TARGET_BOT_NAME = '維特witt 助手'; // 你的 Bot 名字 try { const response = await fetch(`https://discord.com/api/guilds/${SERVER_ID}/widget.json`); const data = await response.json(); // 在 members 陣列中尋找名字相符且狀態不是 offline 的成員 const bot = data.members.find(m => m.username === TARGET_BOT_NAME); const statusElement = document.getElementById('bot-status-display'); if (bot && bot.status !== 'offline') { // 狀態：正常 (Online / Idle / DND) statusElement.innerHTML = `● OPERATIONAL`; console.log("系統回報：Bot 正常運作中"); } else { // 狀態：找不到或離線 statusElement.innerHTML = `○ SYSTEM OFFLINE`; console.log("系統回報：Bot 已離線"); } } catch (error) { console.error("無法取得 Discord 狀態:", error); } } // 每 30 秒自動更新一次，保持數據最新 setInterval(updateBotStatus, 30000); updateBotStatus();
