async function updateBotStatus() { const SERVER_ID = '1330733636219043961'; const TARGET_BOT_NAME = '維特witt 助手'; const badge = document.getElementById('bot-status-badge'); if (!badge) return; const statusText = badge.querySelector('.status-text'); try { // 使用 proxy 或是加上 random 參數強制重新抓取 const url = `https://discord.com/api/guilds/${SERVER_ID}/widget.json?cache=${Math.random()}`; const response = await fetch(url); if (!response.ok) throw new Error('Discord 拒絕連線'); const data = await response.json(); // 在控制台印出來看看，確認有沒有抓到資料 console.log("Discord Data:", data); const bot = data.members.find(m => m.username === TARGET_BOT_NAME); if (bot) { badge.className = 'badge online'; statusText.innerText = '系統運作中'; } else { badge.className = 'badge offline'; statusText.innerText = '服務離線中'; } } catch (error) { console.error("偵測失敗:", error); // 直接把錯誤印在標籤上，讓我們看看到底是什麼錯誤 statusText.innerText = '連線失敗: ' + error.message; badge.className = 'badge offline'; } } // 確保網頁讀取完才執行 window.onload = () => { updateBotStatus(); setInterval(updateBotStatus, 30000); };
